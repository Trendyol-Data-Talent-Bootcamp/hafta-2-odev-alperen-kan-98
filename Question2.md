```
WITH Since1980Trio AS (
  SELECT ATHLETE, SPORT, FIRST_VALUE(YEAR) OVER(PARTITION BY ATHLETE, SPORT ORDER BY ATHLETE, YEAR) AS Ist,
                         LEAD(YEAR,1) OVER(PARTITION BY ATHLETE, SPORT ORDER BY ATHLETE, YEAR) AS IInd,
                         LEAD(YEAR,2) OVER(PARTITION BY ATHLETE, SPORT ORDER BY ATHLETE, YEAR) AS IIIrd
  FROM alperen_kan.summer_medals
  WHERE YEAR >= 1980
  ORDER BY ATHLETE, SPORT)
SELECT ATHLETE FROM Since1980Trio WHERE (Ist+IInd+IIIrd)/3 = IInd AND Ist != IInd;
```
