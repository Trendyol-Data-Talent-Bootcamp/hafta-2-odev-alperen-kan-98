```
WITH Since1980 AS(
  SELECT SPORT, COUNTRY, COUNT(1) AS TOTALMEDALS, ROW_NUMBER() OVER(PARTITION BY SPORT ORDER BY COUNT(1) DESC) AS PLACE
  FROM alperen_kan.summer_medals
  WHERE YEAR > 1980
  GROUP BY SPORT, COUNTRY
  ORDER BY SPORT, TOTALMEDALS DESC)
SELECT * FROM Since1980 WHERE PLACE = 1 OR PLACE = 3 OR PLACE = 5
```
